##database structure
CREATE TABLE `TaskDatabase` (
	`TaskID` INT unsigned NOT NULL AUTO_INCREMENT,
	`UserID` INT NOT NULL,
	`TaskTitle` VARCHAR(255) CHARACTER SET ascii COLLATE ascii_bin NOT NULL,
	`TaskDescription` TEXT CHARACTER SET ascii COLLATE ascii_bin NOT NULL DEFAULT '',
	`TaskType` TINYINT unsigned NOT NULL DEFAULT '0',
	`DueDate` DATE NOT NULL,
	PRIMARY KEY (`TaskID`)
);
